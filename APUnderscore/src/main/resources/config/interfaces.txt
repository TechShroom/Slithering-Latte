# based off of https://docs.python.org/3/reference/datamodel.html


# all objects
# id is special cased to a static lookup
# type is mapped to getClass
-n Deletable -p __del__ -t void;
# str is mapped to toString
-n Representable -p __repr__ -t String;
-n ConvertableToBytes -p __bytes__ -t byte[];
-n Formattable -p __format__ -t String;
-n ComparableLT -p __lt__ -t boolean;
-n ComparableLE -p __le__ -t boolean;
# eq is mapped to equals
-n ComparableNE -p __ne__ -t boolean;
-n ComparableGT -p __gt__ -t boolean;
-n ComparableGE -p __ge__ -t boolean;
# hash is mapped to hashCode
-n ConvertableToBool -p __bool__ -t boolean;
-n PythonObject -p object -s Deletable,Representable,ConvertableToBytes,
                             ComparableMixin<T>,ConvertableToBool -g T --share-generic;

# callable objects
-n DocumentationAttribute -p __doc__ -w -t String;
-n NameAttribute -p __name__ -w -t String;
-n QualifiedNameAttribute -p __qualname__ -w -t String;
-n ModuleNameAttribute -p __module__ -w -t String;
-n DefaultArgsAttribute -p __defualts__ -w -t Tuple;
-n CompiledCodeAttribute -p __code__ -w -t String;
-n GlobalsAttribute -p __global__ -t java.util.Map -g String,Object;
-n AttributeDictionaryAttribute -p __dict__ -w -t java.util.Map -g String,Object;
-n ClosureVariablesAttribute -p __closure__ -t Tuple;
-n ParameterAnnotationsAttribute -p __annotations__ -w -t java.util.Map -g String,Object;
-n KeywordDefaultsAttribute -p __kwdefaults__ -w -t java.util.Map -g String,Object;
-n CallableType -s DocumentationAttribute,NameAttribute,QualifiedNameAttribute,
                   ModuleNameAttribute,DefaultArgsAttribute,CompiledCodeAttribute,
                   GlobalsAttribute,AttributeDictionaryAttribute,
                   ClosureVariablesAttribute,ParameterAnnotationsAttribute,
                   KeywordDefaultsAttribute;

# instance methods, or bound methods (really bound callables)
-n BoundSelfParameterAttribute -p __self__ -t Object;
-n BoundFunctionAttribute -p __func__ -t Object; # actually Function
-n BoundCallableType -s BoundSelfParameterAttribute,BoundFunctionAttribute,CallableType;

# builtins
-n BuiltinMethodType -s DocumentationAttribute,NameAttribute,ModuleNameAttribute;

# bound builtins
-n BoundBuiltinMethodType -s BuiltinMethodType,BoundSelfParameterAttribute;

# classes...sorta?
# __new__ is static
# __init__ is mapped to the constructor

# callable classes
# covered by methods...probably?

# modules
-n FileSource -p __file__ -t String;
-n Module -s AttributeDictionaryAttribute,NameAttribute,DocumentationAttribute,FileSource;
